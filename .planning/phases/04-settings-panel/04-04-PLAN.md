**getDefaultRound helper (module-level):**
```typescript
function getDefaultRound(index: number): RoundConfig {
  const defaults = [
    { interviewer: 'Wayne', theme: 'General',   voice: 'Wayne' },
    { interviewer: 'Atlas', theme: 'Technical', voice: 'Atlas' },
    { interviewer: 'Nova',  theme: 'Cultural',  voice: 'Nova'  },
  ];
  const d = defaults[index] ?? defaults[0];
  return { round: index + 1, theme: d.theme, duration: 30, voice: d.voice, avatarEnabled: true };
}
```

**handleRoundCountChange:**
```typescript
function handleRoundCountChange(n: number) {
  setRoundCount(n);
  setRounds(prev => {
    if (n > prev.length) {
      return [...prev, ...Array.from({ length: n - prev.length }, (_, i) => getDefaultRound(prev.length + i))];
    }
    return prev.slice(0, n);
  });
}
```
