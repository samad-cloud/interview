**Build generation params helper** (inline, not exported):
```typescript
function buildGenParams(state: WizardState) {
  const salaryStr = (state.salaryMin || state.salaryMax)
    ? `${state.salaryCurrency} ${state.salaryMin || '?'}–${state.salaryMax || '?'}/${state.salaryPeriod}`
    : 'Competitive';
  return {
    title: state.title,
    salary: salaryStr,
    location: state.location,
    employmentType: state.employmentType.replace(/_/g, ' '),
    experienceLevel: state.experienceMin ? `${state.experienceMin}–${state.experienceMax || '+'} years` : undefined,
    keySkills: state.skillsMustHave.join(', ') || undefined,
    mustHave: state.skillsMustHave.join(', ') || undefined,
    niceToHave: state.skillsNiceToHave.join(', ') || undefined,
  };
}
```

**handleGenerate:**
```typescript
async function handleGenerate() {
  setIsGenerating(true);
  setError(null);
  try {
    const result = await generateJobDescription(buildGenParams(props.state));
    props.onChange({ generatedDescription: result.description, refinePrompt: '' });
    setCitations(result.citations);
  } catch (e) {
    setError('Generation failed. Please try again.');
  } finally {
    setIsGenerating(false);
  }
}
```

**handleRefine:**
```typescript
async function handleRefine() {
  if (!props.state.refinePrompt.trim()) return;
  setIsRefining(true);
  setError(null);
  try {
    const refined = await refineJobDescription(props.state.generatedDescription, props.state.refinePrompt);
    props.onChange({ generatedDescription: refined, refinePrompt: '' });
  } catch (e) {
    setError('Refinement failed. Please try again.');
  } finally {
    setIsRefining(false);
  }
}
```
